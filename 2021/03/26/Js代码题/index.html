js 将多层嵌套的数组转换为一层数组

let list = [1, [2, [3, [4]], 5]];

Array.prototype.transfer = function (level) {
  let list = this;
  const flat = (list) => {
    return list.reduce((prev, next) => {
      next = Array.isArray(next) ? next : [next];
      return [...prev, ...next];
    }, []);
  };
  while (level > 0) {
    list = flat(list);
    level--;
  }
  return list;
};

let list = this调用时候的数组，在原型链上的东西，使用时即调用本身。
